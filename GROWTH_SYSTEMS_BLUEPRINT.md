### Critical Events to Track

**1. Scroll Depth Tracking**
```javascript
let maxScroll = 0;
window.addEventListener('scroll', () => {
  const scrollPct = Math.round((window.scrollY / (document.body.scrollHeight - window.innerHeight)) * 100);
  if (scrollPct > maxScroll) {
    maxScroll = scrollPct;
    if ([25, 50, 75, 90, 100].includes(scrollPct)) {
      gtag('event', 'scroll_depth', {
        value: scrollPct,
        event_category: 'engagement',
        scroll_percent: scrollPct
      });
    }
  }
});
```

**2. CTA Click Tracking**
```javascript
// Add data-cta attribute to all buttons
document.addEventListener('click', (e) => {
  const cta = e.target.closest('[data-cta]');
  if (cta) {
    gtag('event', 'cta_click', {
      event_category: 'conversion',
      event_label: cta.dataset.cta,
      button_text: cta.innerText
    });
  }
});
```

**3. Form Submission Tracking**
```javascript
document.addEventListener('submit', (e) => {
  const form = e.target;
  if (form.dataset.form) {
    gtag('event', 'form_submit', {
      event_category: 'lead',
      event_label: form.dataset.form,
      form_id: form.id
    });
  }
});
```

**4. Engagement Time Tracking**
```javascript
let startTime = Date.now();
window.addEventListener('beforeunload', () => {
  const engagementTime = Math.round((Date.now() - startTime) / 1000);
  gtag('event', 'engagement_time', {
    value: engagementTime,
    event_category: 'engagement'
  });
});
```
